def suma_lista(lista):
    suma = 0
    for elemento in lista:
        suma += elemento
    return suma

"""
4. Contar dígitos
Dado un número entero, retornar cuántos dígitos tiene usando recursión.
12345 -> 5
213 -> 3
1000000 -> 7
"""
# Definimos la funciòn contar_digitos
# Definimos un caso base
# Definimos un movimiento de avance -> va con la llamada recursiva
def contar_digitos(numero):
    if numero == 0: # Caso Base
        return 0 # Caso Base
    return 1 + contar_digitos(numero // 10)

# -> 1 + 1 + 1 + 1 + contar_digitos(1) 
# -> 1 + 1 + 1 + 1 + 1 + contar_digitos(0) 
# -> 1 + 1 + 1 + 1 + 1 + 0 -> 5
